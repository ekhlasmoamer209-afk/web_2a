function calculateStatus($grade) {
    if ($grade >= 90) {
        return "ممتاز";
    } elseif ($grade >= 80) {
        return "جيد جدا";
    } elseif ($grade >= 70) {
        return "جيد";
    } elseif ($grade >= 60) {
        return "مقبول";
    } else {
        return "راسب";
    }
}
